Как потом дополнять этот файл

Вся база лежит ниже в JavaScript-массивах. Просто открывай этот HTML в редакторе и добавляй новые записи по образцу. После сохранения обнови страницу в браузере.

Куда добавлять новые локации
const LOCATIONS = [ { id: "unikalnyy-id", name: "Русское название локации", region: "Регион / зона", confidence: "high | medium | low", notes: "Короткая заметка", sources: ["Источник 1", "Источник 2"], unlocks: [ { object: "Русское название объекта", type: "станция | генератор | ящик | защита | транспорт | расходник | оружие | инфраструктура | другое", notes: "Что именно найдено / где именно", confidence: "high | medium | low" } ] } ];
Куда добавлять новые рецепты
const RECIPES = [ { id: "recept-id", name: "Русское название технологии", confidence: "high | medium | low", scenario: "Все / Manibus / Endless Dream / Неизвестно", notes: "Комментарий", ingredients: [ { item: "Название ресурса", qty: 1 }, { item: "Название ресурса", qty: 2 } ], sources: ["Источник 1"] } ];
